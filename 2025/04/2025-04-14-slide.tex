\section{FLP写作与实验}
\begin{frame}{FLP部分撰写与实验}
  \begin{block}{函数级并行（FLP）}
    \begin{itemize}
      \item FLP为Thread-Adaptive架构的重要组成部分
      \item 主要策略：
        \begin{itemize}
          \item WOTS$^+$并行化：$l$个哈希链并发计算
          \item FORS并行化：$k \times 2^a$密钥元素并行生成
          \item Hypertree并行化：跨$d$层多Merkle树并发构建
        \end{itemize}
      \item 合理利用合并内存访问与共享内存
    \end{itemize}
  \end{block}
\end{frame}
